participant App
participant Server

App->Server:loginAdmin(data)
App<--Server:Response
Admin<--App:Show dashboard

opt Create bug report
Admin->>App:Create bug report
App->>Server:addBugReport(bugInfo)
end

opt Add doctor
Admin->App:Add doctor
Admin<--App:Show new doctor form

App->>Server:createDoctor(doctorData)
end 

opt View vaccination history
Admin->App:View vaccination history
App->Server:getVaccinations(filters)

Admin<--App:Show report
end 

opt View patients
Admin->App:View patients
App->Server:getPatients()

Admin<--App:Show patients
end

opt Edit patient
Admin->App:Edit patient
Admin<--App:Show edit form
Admin->>App:Submit new patient data
App->>Server:editPatient(patientId, newPatientData)
end
